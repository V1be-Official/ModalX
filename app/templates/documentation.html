HOW TO USE IT

1. Download CSS and JS files from GitHub

https://github.com/V1be-Official/ModalX

2. Choose which modal window you need to use

ALERT

var objAlert = {
titleText: "Подтвердите действия!",
text: "Добро пожаловать!",
className: "center-x-top",
duration: 400,
durationOut: 3000,
type: "primary",
buttonText: "Хорошо",
button: true,
}

CONFIRM

var objConfirm = {
titleText: "Подтвердите действия!",
text: "Добро пожаловать!",
className: "center-x-top",
duration: 400,
durationOut: 3000,
type: "primary",
buttonTextFalse: "Нет",
buttonTextTrue: "Да"
}

PROMPT

var objPrompt = {
titleText: "Подтвердите действия!",
text: "Добро пожаловать!",
className: "center-x-top",
duration: 400,
durationOut: 3000,
type: "primary",
buttonText: "Хорошо",
inputText: "Type your name"
}

3. Calling objects

ALERT

var modal = new ModalX();
modal.alert(objAlert);

CONFIRM

var modal = new ModalX();
modal.confirm(objConfirm, function(result) {
console.log(result);
});

PROMPT

var modal = new ModalX();
modal.prompt(objPrompt,function(result) {
console.log(result);
});
